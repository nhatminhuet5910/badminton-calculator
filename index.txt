    function formatMoney(amount) {
        return amount.toLocaleString('vi-VN', { style: 'currency', currency: 'VND' });
    }

    function calculateCost() {
        let court = Number(document.getElementById('courtFee').value) || 0;
        let water = Number(document.getElementById('waterFee').value) || 0;
        let other = Number(document.getElementById('otherFee').value) || 0;
        let shuttle = Number(document.getElementById('shuttleFee').value) || 0;
        
        let males = Number(document.getElementById('maleCount').value) || 0;
        let females = Number(document.getElementById('femaleCount').value) || 0;

        let totalPeople = males + females;

        if (totalPeople === 0) {
            alert("Vui l√≤ng nh·∫≠p s·ªë ng∆∞·ªùi ch∆°i!");
            return;
        }

        let commonTotal = court + water + other;
        let commonPerPerson = commonTotal / totalPeople;

        let shuttlePerMale = 0;
        let shuttlePerFemale = 0;

        if (males > 0 && females > 0) {
            let totalShuttleMale = shuttle * (2/3);
            shuttlePerMale = totalShuttleMale / males;

            let totalShuttleFemale = shuttle * (1/3);
            shuttlePerFemale = totalShuttleFemale / females;
        } else if (males > 0 && females === 0) {
            shuttlePerMale = shuttle / males;
        } else if (females > 0 && males === 0) {
            shuttlePerFemale = shuttle / females;
        }

        let finalMalePrice = commonPerPerson + shuttlePerMale;
        let finalFemalePrice = commonPerPerson + shuttlePerFemale;

        let totalCost = commonTotal + shuttle;
        
        document.getElementById('displayTotal').innerText = formatMoney(totalCost);
        document.getElementById('maleShare').innerText = formatMoney(finalMalePrice);
        document.getElementById('femaleShare').innerText = formatMoney(finalFemalePrice);

        document.getElementById('resultArea').style.display = 'block';
        
        document.getElementById('resultArea').scrollIntoView({ behavior: 'smooth' });
    }
